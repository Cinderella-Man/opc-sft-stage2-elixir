```elixir
defmodule CountOccurrences do
  @doc """
  Counts the number of times a target value appears in a list using strict equality (`===`).

  ## Examples

      iex> CountOccurrences.count_occurrences([1, 2, 3, 2], 2)
      2

      iex> CountOccurrences.count_occurrences([1.0, 2, 3], 1)
      0
  """
  @spec count_occurrences(list(), any()) :: non_neg_integer()
  def count_occurrences(list, target) do
    Enum.count(list, &(&1 === target))
  end
end
```
