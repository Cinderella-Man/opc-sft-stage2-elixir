defmodule Solution do
  @doc """
  Finds the maximum number that can be formed by rearranging the digits of a given non-negative integer.

  ## Examples

      iex> max_number(213)
      321

      iex> max_number(120)
      210
  """
  @spec max_number(non_neg_integer()) :: non_neg_integer()
  def max_number(n) when n >= 0 do
    n
    |> Integer.digits()
    |> Enum.sort(:desc)
    |> Enum.reduce(0, fn digit, acc ->
      acc * 10 + digit
    end)
  end
end
